Traditional Loops

            var i,
                array = generateSomeArray(),
                len = array.length,


            for(i = 0; i < len; i++)
              doFancyStuff(array[i]);

            /* OR */

            i = len;
            while(i--)
              doFancyStuff(array[i]);

Functional Loops

            var array = generateSomeArray(),
                result;

            array.forEach(doFancyStuff);

            /* OR */

            result = array.map(doFancyStuff);

            /* OR */

            result = array.filter(doFancyStuff);

            /* OR EVEN */

            array.reduce(function(prev, curr, i){ return prev + curr; });
